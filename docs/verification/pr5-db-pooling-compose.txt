# PgBouncer Docker Compose Configuration
# File: docker-compose.pgbouncer.yml

version: '3.8'

services:
  pgbouncer:
    image: edoburu/pgbouncer:latest
    container_name: uberfoods_pgbouncer
    environment:
      - DATABASE_URL=postgresql://postgres:postgres123@postgres:5432/uberfoods
      - POOL_MODE=transaction
      - MAX_CLIENT_CONN=100
      - DEFAULT_POOL_SIZE=10
      - MIN_POOL_SIZE=2
      - RESERVE_POOL_SIZE=5
      - RESERVE_POOL_TIMEOUT=5
      - MAX_DB_CONNECTIONS=50
      - MAX_USER_CONNECTIONS=50
    ports:
      - "6432:5432"  # Expose PgBouncer on port 6432
    depends_on:
      - postgres
    networks:
      - uberfoods_network
    healthcheck:
      test: ["CMD", "pg_isready", "-h", "localhost", "-p", "5432", "-U", "postgres"]
      interval: 10s
      timeout: 5s
      retries: 5

networks:
  uberfoods_network:
    external: true
